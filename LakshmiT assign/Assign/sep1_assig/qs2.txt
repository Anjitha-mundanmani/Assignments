 ip : 192.0.4.8   subnet mask: 255.255.255.252
        11000000.00000000.00000100.000010 00  192.0.4.8
        11111111.11111111.11111111.111111 00  255.255.255.252
     -------------------------------------------
	11000000.00000000.00000100.000010 00 = network address =192.0.4.8 

	

	11000000.00000000.00000100.000010 11 broadcast address =192.0.4.11 
	
	Available hosts:	4-2 = 2
	IP addr range:	192.0.4.9 - 192.0.4.10

       -------------------------------------------

	IP addr range:	10.1.4.1 - 10.1.5.254
